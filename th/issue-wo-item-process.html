การตรวจสอบสิทธิ์
API นี้ต้องการสิทธิ์โมดูล material (ระบบจัดการวัสดุ) หากไม่มีสิทธิ์จะส่งคืนข้อผิดพลาด

พารามิเตอร์คำขอ (ระดับนอก)

ฟิลด์ประเภทจำเป็นคำอธิบาย
doc_noStringYเลขที่เอกสาร
create_dateStringYวันที่สร้าง
statusStringYสถานะการดำเนินการ (1/2/3/4)
remarkStringNหมายเหตุ
wo_item_detailArrayYอาร์เรย์รายละเอียดวัสดุใบสั่งผลิต

พารามิเตอร์คำขอ (รายละเอียด wo_item_detail)

ฟิลด์ประเภทจำเป็นคำอธิบาย

wo_noStringYเลขที่ใบสั่งผลิต
item_typeStringYระดับวัสดุ (0=สินค้าสำเร็จรูป/1=วัตถุดิบ)
item_noStringYรหัสสินค้า (รหัสวัสดุ)
qtyNumericYจำนวนที่เปลี่ยนแปลง (ต้อง > 0)
unit_noStringNหน่วย
input_datetimeStringNเวลาบันทึก
warehouse_noStringNคลังสินค้า
location_noStringNตำแหน่งจัดเก็บ
lot_noStringNเลขที่ล็อต
expiry_dateStringNวันหมดอายุ
remarkStringNหมายเหตุ
positive_negativeStringYทิศทาง (-1=เบิก/อื่นๆ=คืน)
issue_to_typeStringNประเภทการเบิก
op_noStringNรหัสกระบวนการ

คำอธิบายสถานะการดำเนินการ

รหัสสถานะคำอธิบายการคำนวณจำนวน
1เพิ่มการเบิกdemand_quantity + qty
2ลบการเบิกdemand_quantity - qty
3เพิ่มการคืนdemand_quantity - qty
4ลบการคืนdemand_quantity + qty
การตรวจสอบจำนวน
เมื่อดำเนินการหักลบ (สถานะ 2, 3) หาก resultQty < 0 จะเกิดข้อผิดพลาด:
JAVA_0119_00057: จำนวนที่ต้องเบิกไม่เพียงพอ ไม่สามารถดำเนินการหักลบได้

ตรรกะการประมวลผล

กระบวนการทางธุรกิจ
  1. ตรวจสอบสิทธิ์โมดูล material
  2. ตรวจสอบว่า status ต้องเป็น 1/2/3/4
  3. ตรวจสอบว่า wo_no (เลขที่ใบสั่งผลิต) มีอยู่ใน MODETAIL
  4. ค้นหา material_requisition เพื่อรับ demand_quantity ปัจจุบัน
  5. คำนวณ resultQty ใหม่ตาม status
  6. เพิ่มระเบียน material_requisition
  7. เพิ่ม/อัปเดต material_apply_basis (ส่วนหัวใบเบิก-คืน)
  8. เพิ่ม material_apply_detail (รายละเอียดใบเบิก-คืน)

SQL อัปเดตฐานข้อมูล

1. ตรวจสอบว่าใบสั่งผลิตมีอยู่

SELECT 1 FROM MODETAIL WITH (NOLOCK) WHERE CMOID = :wo_no

2. ค้นหาจำนวนที่ต้องเบิกปัจจุบัน

SELECT TOP 1 ISNULL(demand_quantity, 0) AS demand_quantity
FROM material_requisition WITH (NOLOCK)
WHERE modetail_cmoid = :wo_no
  AND material_id = :item_no
ORDER BY serial DESC

3. INSERT material_requisition (รายละเอียดวัสดุใบสั่งผลิต)

INSERT INTO material_requisition (
    serial, modetail_cmoid, material_type, material_id,
    opid, erp_wsid, material_batch, remark,
    demand_quantity, withdrawn_quantity, mr01
) VALUES (
    :serial,           -- MAX(serial) + 1
    :wo_no,            -- เลขที่ใบสั่งผลิต
    :item_type,        -- ประเภทวัสดุ (1=สินค้าสำเร็จรูป/2=วัตถุดิบ)
    :item_no,          -- รหัสสินค้า
    :op_no,            -- รหัสกระบวนการ
    '',                -- รหัสสถานีงาน
    :lot_no,           -- เลขที่ล็อต
    :remark,           -- หมายเหตุ
    :resultQty,        -- จำนวนที่ต้องเบิกหลังคำนวณ
    0,                 -- จำนวนที่เบิกแล้ว
    N'{เวลาปัจจุบัน}'    -- เวลาสร้าง
)

4. INSERT material_apply_basis (ส่วนหัวใบเบิก-คืน)

IF NOT EXISTS (
    SELECT 1 FROM material_apply_basis
    WHERE material_order_id = :doc_no
)
INSERT INTO material_apply_basis (
    material_order_id, material_order_type, factory_id,
    order_source, material_order_status,
    creator, create_time, modifier, modify_time
) VALUES (
    :doc_no,           -- เลขที่เอกสาร
    :positive_negative, -- 1=เบิก/2=คืน
    '001',             -- รหัสโรงงาน
    '2',               -- แหล่งที่มา (ERP)
    '1',               -- สถานะ
    'ERP_SYSTEM',
    N'{เวลาปัจจุบัน}',
    'ERP_SYSTEM',
    N'{เวลาปัจจุบัน}'
)

5. INSERT material_apply_detail (รายละเอียดใบเบิก-คืน)

IF NOT EXISTS (
    SELECT 1 FROM material_apply_detail
    WHERE material_order_id = :doc_no
      AND material_order_seq = :seq
)
INSERT INTO material_apply_detail (
    material_order_id, material_order_seq, mo_id, material_seq,
    material_id, apply_qty, receive_qty, op_id, ws_id,
    material_type, standard_quantity,
    creator, create_time, modifier, modify_time
) VALUES (
    :doc_no,           -- เลขที่เอกสาร
    :seq,              -- ลำดับ
    :wo_no,            -- เลขที่ใบสั่งผลิต
    :serial,           -- ลำดับวัสดุ
    :item_no,          -- รหัสสินค้า
    :qty,              -- จำนวนขอเบิก
    :qty,              -- จำนวนเบิก
    :op_no,            -- รหัสกระบวนการ
    '',                -- รหัสสถานีงาน
    :item_type,        -- ประเภทวัสดุ
    :qty,              -- จำนวนมาตรฐาน
    'ERP_SYSTEM',
    N'{เวลาปัจจุบัน}',
    'ERP_SYSTEM',
    N'{เวลาปัจจุบัน}'
)

รหัสข้อผิดพลาด

รหัสข้อความคำอธิบาย
JAVA_0119_00059สถานะต้องไม่ว่างและต้องเป็น 1/2/3/4การตรวจสอบพารามิเตอร์ status ล้มเหลว
JS_0097_00005เลขที่ใบสั่งผลิตต้องไม่ว่างwo_no เป็นค่าว่าง
JAVA_0119_00058ระดับวัสดุต้องไม่ว่างและต้องเป็น 0 หรือ 1การตรวจสอบ item_type ล้มเหลว
JS_0085_00004รหัสสินค้าต้องไม่ว่างitem_no เป็นค่าว่าง
JAVA_0119_00060จำนวนที่เปลี่ยนแปลงต้องมากกว่า 0การตรวจสอบ qty ล้มเหลว
JSP_0119_00009ไม่พบใบสั่งผลิตไม่พบในตาราง MODETAIL
JAVA_0119_00057จำนวนที่ต้องเบิกไม่เพียงพอ ไม่สามารถหักลบได้จำนวนหลังหักลบเป็นค่าลบ
eMES_Material_007โมดูลจัดการวัสดุไม่ได้รับอนุญาตโมดูล material
