ฟิลด์การตอบกลับ

ฟิลด์ ERPฟิลด์ MESคำอธิบาย
employee_noMV001รหัสพนักงาน (Primary Key)
employee_nameMV002ชื่อพนักงาน
department_noMV004รหัสแผนก
emailMV020อีเมล
resignation_dateMV022วันที่ลาออก
tran_statusTRAN_STATUSสถานะธุรกรรม

SQL อัปเดตฐานข้อมูล

ตรวจสอบเพิ่มหรือแก้ไข

SELECT COUNT(1) FROM ERP_CMSMV WHERE MV001 = N'{employee_no}'
-- count > 0 → UPDATE, count = 0 → INSERT

INSERT (เพิ่มพนักงาน)

INSERT INTO ERP_CMSMV (MV001, MV002, MV004, MV020, MV022)
VALUES (
    :userId,
    :userName,
    '',             -- ค่าเริ่มต้นแผนกว่าง
    '',             -- ค่าเริ่มต้นอีเมลว่าง
    ''              -- ค่าเริ่มต้นวันที่ลาออกว่าง
)

UPDATE (แก้ไขพนักงาน)

UPDATE ERP_CMSMV SET
    MV002 = :userName
WHERE MV001 = :userId

SFT_PRINCIPAL (การเชื่อมโยงบัญชีระบบ)

-- เพิ่มบัญชีระบบ
INSERT INTO {EMESSYS}..SFT_PRINCIPAL (
    AUTH_ID, AUTH_NAME, AUTH_PASSWORD, USEDSTATE,
    AUTH_DESCRIPTION, Creator, CreateDate, PHONE, MAIL, WECHAT
) VALUES (
    :userId, :userName, :encryptPassWord, :status,
    :description, 'ESB', GETDATE(), :phone, :mail, :weChat
)

-- แก้ไขบัญชีระบบ
UPDATE {EMESSYS}..SFT_PRINCIPAL SET
    AUTH_NAME = :userName,
    AUTH_PASSWORD = :encryptPassWord,
    USEDSTATE = :status,
    AUTH_DESCRIPTION = :description,
    PHONE = :phone,
    MAIL = :mail,
    WECHAT = :weChat
WHERE AUTH_ID = :userId
