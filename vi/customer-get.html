INSERT (Them moi khach hang)

INSERT INTO CUSTOMER (
    ID, SHOTNAME, NAME, TEL, FAX, EMAIL,
    AREA_ID, AREA_NAME, ADDRESS,
    CHECKMAINTAINNO, LASTMAINTAINUSER, LASTMAINTAINDATETIME
) VALUES (
    :CUSTOMERID,
    :SHOTNAME,
    :CUSTOMER_NAME,
    :TEL,
    :FAX,
    :EMAIL,
    :AREA_ID,
    :AREA_NAME,
    :ADDRESS,
    1,
    :Creator,
    GETDATE()
)

UPDATE (Cap nhat khach hang)

UPDATE CUSTOMER SET
    SHOTNAME = :SHOTNAME,
    NAME = :CUSTOMER_NAME,
    TEL = :TEL,
    FAX = :FAX,
    EMAIL = :EMAIL,
    AREA_ID = :AREA_ID,
    AREA_NAME = :AREA_NAME,
    ADDRESS = :ADDRESS
WHERE ID = :CUSTOMERID
