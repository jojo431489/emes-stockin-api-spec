9. SQL Cập nhật cơ sở dữ liệu

9.1 Kiểm tra lệnh sản xuất tồn tại

-- Kiểm tra trước khi thêm, nếu tồn tại sẽ trả về JS_0126_00061
SELECT COUNT(1) FROM MODETAIL WHERE CMOID = N'{wo_no}'

9.2 INSERT MODETAIL (Thêm bảng chính lệnh sản xuất)

INSERT INTO MODETAIL (
    CMOID, MO004, MO005, ITEMID, MO021, MO022, UNIT,
    PLANPROCESSST, DUEDATETIME, QTY, STATUS, DOID,
    SEQUENCE, FACTORYID, WAREHOUSEID, DESCRIPTION,
    MO008, MO012, MO016, MO032, PLANRELEASEQTY,
    FLAG, LASTMAINTAINUSER, LASTMAINTAINDATETIME
) VALUES (
    :CMOID,              -- Số lệnh sản xuất (Loại-Số)
    :MO004,              -- Loại lệnh
    :MO005,              -- Số lệnh
    :ITEMID,             -- Mã sản phẩm
    :MO021,              -- Tên sản phẩm
    :MO022,              -- Quy cách
    :UNIT,               -- Đơn vị
    :PLANPROCESSST,      -- Ngày bắt đầu dự kiến
    :DUEDATETIME,        -- Ngày hoàn thành dự kiến
    :QTY,                -- Sản lượng dự kiến
    1,                   -- STATUS = 1 (Chưa sản xuất)
    :DOID,               -- Số đơn hàng
    :SEQUENCE,           -- Mục đơn hàng
    :FACTORYID,          -- Mã nhà máy
    :WAREHOUSEID,        -- Mã kho
    :DESCRIPTION,        -- Ghi chú
    1,                   -- MO008 = 1
    :MO012,              -- Ngày lập đơn
    0,                   -- MO016 = 0
    :MO032,              -- Mức ưu tiên (1=Gấp)
    :QTY,                -- Số lượng phát hành dự kiến = Sản lượng dự kiến
    1,                   -- FLAG số phiên bản
    :userid,
    N'{Thời gian hiện tại}'
)

9.3 INSERT LOT (Thêm bản ghi lô)

INSERT INTO LOT (
    ID, TYPE, MOID, ITEMID, LOTSIZE, STATUS, UNIT,
    QTYPER, LOCATION, ERP_OPSEQ, ERP_OPID, ERP_WSID,
    LOT004, PKQTY, PKQTYPER, LOT007, LOT011,
    LASTMAINTAINUSER, LASTMAINTAINDATETIME
) VALUES (
    :CMOID,              -- Số lô = Số lệnh sản xuất
    1,                   -- TYPE = 1 (Lô kế hoạch)
    :CMOID,              -- MOID = Số lệnh sản xuất
    :ITEMID,             -- Mã sản phẩm
    :QTY,                -- LOTSIZE = Sản lượng dự kiến
    0,                   -- STATUS = 0 (Chờ phát hành)
    :UNIT,               -- Đơn vị
    1,                   -- QTYPER = 1
    'release',           -- LOCATION = 'release'
    '', '', '',          -- Trường quy trình ERP (trống)
    1, 0, 0,             -- LOT004, PKQTY, PKQTYPER
    '0', 'N',            -- LOT007, LOT011
    :userid,
    N'{Thời gian hiện tại}'
)

9.4 Khởi tạo quy trình SFT_OP_REALRUN (Đọc từ SFT_BOMMF)

-- Kiểm tra sản phẩm có lộ trình tiêu chuẩn không
SELECT ME007 FROM SFT_BOMME, MODETAIL, ITEM
WHERE ITEM.ID = MODETAIL.ITEMID
  AND CMOID = N'{wo_no}'
  AND RTRIM(ME001) = ITEM.STDROUTEITEMID
  AND RTRIM(ME002) = ITEM.STDROUTEID

-- Nếu có lộ trình, xóa rồi thêm mới
DELETE SFT_OP_REALRUN WHERE ID = N'{wo_no}'

-- Sao chép dữ liệu quy trình từ SFT_BOMMF
INSERT INTO SFT_OP_REALRUN (
    ID, SEQUENCE, OPID, OPDESCRIPTION,
    ARRIVEQTY, OUTQTY, REWORKQTY, DEFECTQTY,
    ERP_OPSEQ, PERLOTWORKTIME, STANDARDEQWORKTIME, STANDARDMANWORKTIME,
    ERP_OPID, ERP_WSID, ERP_FIXEDLEADTIME, ERP_TRANSQTY, ...
)
SELECT
    CMOID,                          -- ID = Số lệnh sản xuất
    0,                              -- SEQUENCE = 0
    RTRIM(MF004)+'---'+RTRIM(MF006), -- OPID = Quy trình+Trạm
    MF008,                          -- OPDESCRIPTION = Mô tả quy trình
    0, 0, 0, 0,                     -- Trường số lượng khởi tạo = 0
    RTRIM(MF003),                   -- ERP_OPSEQ = Thứ tự quy trình
    MF019,                          -- PERLOTWORKTIME = Thời gian mỗi lô
    ...
FROM SFT_BOMMF, MODETAIL, ITEM
WHERE ITEM.ID = MODETAIL.ITEMID
  AND CMOID = N'{wo_no}'
  AND RTRIM(MF001) = ITEM.STDROUTEITEMID
  AND RTRIM(MF002) = ITEM.STDROUTEID

9.5 Ghi vào material_requisition (Vật tư lệnh sản xuất)

Đồng bộ dữ liệu vật tư lệnh sản xuất
API wo.create sẽ ghi trực tiếp dữ liệu vật tư lệnh sản xuất vào bảng material_requisition. 
-- wo.create xử lý dữ liệu vật tư lệnh sản xuất
-- 1. Đầu tiên ghi vào bảng order_material (dữ liệu đơn hàng vật tư gốc)
SELECT COUNT(1) FROM order_material
WHERE material_requisition = N'{material_requisition}'
  AND material_number = N'{material_number}'
  AND serial_number = N'{serial_number}'
-- count > 0 → UPDATE, count = 0 → INSERT

-- INSERT order_material
INSERT INTO order_material (
    material_requisition, material_number, serial_number,
    material_id, material_quantity, craftsmanship,
    batch_number, category, order_number, type,
    batch_description, om01, om02, om03, om04
) VALUES (
    :material_requisition, :material_number, :serial_number,
    :material_id, :material_quantity, :craftsmanship,
    :batch_number, :category, :order_number, :type,
    :batch_description, N'{Thời gian hiện tại}', :om02, :om03, :om04
)

-- 2. Gộp các bản ghi cùng Sản phẩm+Quy trình+Trạm+Lô vật tư+Lệnh, ghi vào material_requisition
SELECT COUNT(1) FROM material_requisition
WHERE modetail_cmoid = N'{Số lệnh sản xuất}'
  AND material_id = N'{Mã vật tư}'
  AND opid = N'{Mã quy trình}'
  AND erp_wsid = N'{Mã trạm}'
  AND material_batch = N'{Lô vật tư}'
  AND (remark IS NULL OR remark = '')
-- count > 0 → UPDATE (cộng dồn số lượng), count = 0 → INSERT

-- INSERT material_requisition (thêm bản ghi vật tư)
INSERT INTO material_requisition (
    serial, modetail_cmoid, material_id,
    opid, erp_wsid, material_batch,
    withdrawn_quantity, mr01
) VALUES (
    :serial,              -- Số thứ tự (MAX(serial) + 1)
    :modetail_cmoid,      -- Số lệnh sản xuất (category-order_number)
    :material_id,         -- Mã vật tư
    :opid,                -- Mã quy trình
    :erp_wsid,            -- Mã trạm
    :material_batch,      -- Lô vật tư
    :withdrawn_quantity,  -- Số lượng xuất thực tế
    N'{Thời gian hiện tại}'  -- mr01
)

-- UPDATE material_requisition (cộng dồn số lượng xuất)
UPDATE material_requisition SET
    withdrawn_quantity = withdrawn_quantity + :Số_lượng_mới
WHERE modetail_cmoid = N'{Số lệnh sản xuất}'
  AND material_id = N'{Mã vật tư}'
  AND opid = N'{Mã quy trình}'
  AND erp_wsid = N'{Mã trạm}'
  AND material_batch = N'{Lô vật tư}'
  AND (remark IS NULL OR remark = '')
