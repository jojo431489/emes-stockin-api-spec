Xuất/Trả nguyên liệu lệnh sản xuất

POST ERP → MES
issue.wo.item.process

ERP thông báo MES về thao tác xuất/trả nguyên liệu của lệnh sản xuất, đồng bộ dữ liệu vật tư, phiếu xuất/trả và chi tiết.

Logic xử lý

Quy trình nghiệp vụ
  1. Kiểm tra quyền module material
  2. Xác nhận status phải là 1/2/3/4
  3. Xác nhận wo_no (số lệnh sản xuất) tồn tại trong MODETAIL
  4. Truy vấn material_requisition để lấy demand_quantity hiện tại
  5. Tính toán resultQty mới theo status
  6. Thêm bản ghi material_requisition
  7. Thêm/Cập nhật material_apply_basis (tiêu đề phiếu xuất/trả)
  8. Thêm material_apply_detail (chi tiết phiếu xuất/trả)

SQL cập nhật cơ sở dữ liệu

1. Kiểm tra lệnh sản xuất tồn tại

SELECT 1 FROM MODETAIL WITH (NOLOCK) WHERE CMOID = :wo_no

2. Truy vấn số lượng yêu cầu hiện tại

SELECT TOP 1 ISNULL(demand_quantity, 0) AS demand_quantity
FROM material_requisition WITH (NOLOCK)
WHERE modetail_cmoid = :wo_no
  AND material_id = :item_no
ORDER BY serial DESC

3. INSERT material_requisition (chi tiết vật tư lệnh sản xuất)

INSERT INTO material_requisition (
    serial, modetail_cmoid, material_type, material_id,
    opid, erp_wsid, material_batch, remark,
    demand_quantity, withdrawn_quantity, mr01
) VALUES (
    :serial,           -- MAX(serial) + 1
    :wo_no,            -- Số lệnh sản xuất
    :item_type,        -- Loại vật tư (1=Thành phẩm/2=Nguyên liệu)
    :item_no,          -- Mã sản phẩm
    :op_no,            -- Mã quy trình
    '',                -- Mã trạm làm việc
    :lot_no,           -- Số lô
    :remark,           -- Ghi chú
    :resultQty,        -- Số lượng yêu cầu sau tính toán
    0,                 -- Số lượng đã xuất
    N'{Thời gian hiện tại}'  -- Thời gian tạo
)

4. INSERT material_apply_basis (tiêu đề phiếu xuất/trả)

IF NOT EXISTS (
    SELECT 1 FROM material_apply_basis
    WHERE material_order_id = :doc_no
)
INSERT INTO material_apply_basis (
    material_order_id, material_order_type, factory_id,
    order_source, material_order_status,
    creator, create_time, modifier, modify_time
) VALUES (
    :doc_no,           -- Số chứng từ
    :positive_negative, -- 1=Xuất/2=Trả
    '001',             -- Mã nhà máy
    '2',               -- Nguồn (ERP)
    '1',               -- Trạng thái
    'ERP_SYSTEM',
    N'{Thời gian hiện tại}',
    'ERP_SYSTEM',
    N'{Thời gian hiện tại}'
)

5. INSERT material_apply_detail (chi tiết phiếu xuất/trả)

IF NOT EXISTS (
    SELECT 1 FROM material_apply_detail
    WHERE material_order_id = :doc_no
      AND material_order_seq = :seq
)
INSERT INTO material_apply_detail (
    material_order_id, material_order_seq, mo_id, material_seq,
    material_id, apply_qty, receive_qty, op_id, ws_id,
    material_type, standard_quantity,
    creator, create_time, modifier, modify_time
) VALUES (
    :doc_no,           -- Số chứng từ
    :seq,              -- Số thứ tự
    :wo_no,            -- Số lệnh sản xuất
    :serial,           -- Số serial vật tư
    :item_no,          -- Mã sản phẩm
    :qty,              -- Số lượng yêu cầu
    :qty,              -- Số lượng nhận
    :op_no,            -- Mã quy trình
    '',                -- Mã trạm làm việc
    :item_type,        -- Loại vật tư
    :qty,              -- Số lượng tiêu chuẩn
    'ERP_SYSTEM',
    N'{Thời gian hiện tại}',
    'ERP_SYSTEM',
    N'{Thời gian hiện tại}'
)
